🎯 מה אנחנו בונים?

הפרויקט הוא בדיקת End-to-End (E2E) על האתר tutorialsninja.com/demo – אתר קניות דמו (OpenCart). הבדיקה מדמה משתמש אמיתי שמחפש מוצרים, מסנן לפי מחיר, מוסיף לסל ומאמת את הסכום.

1

הזדהות (Login)

כניסה לאתר עם שם משתמש וסיסמא. אם המשתמש לא קיים – נרשום אחד אוטומטית.

2

חיפוש לפי מחיר – search_items_by_name_under_price

מחפשים מוצרים לפי מילת מפתח ומחזירים רשימת URLs של מוצרים שמחירם <= maxPrice.

3

הוספה לסל – add_items_to_cart

עוברים על כל URL, נכנסים לדף המוצר, בוחרים וריאנטים (צבע/מידה) ולוחצים "Add to Cart".

4

אימות סכום – assert_cart_total_not_exceeds

נכנסים לעגלת הקניות, קוראים את הסכום הכולל ומאמתים שהוא ≤ budgetPerItem × מספר פריטים.

🧠 מושגי יסוד – POM / OOP / Data-Driven

📐 Page Object Model (POM)

במקום לכתוב page.locator("#search input").fill("Apple") ישירות בתוך הבדיקה, אנחנו יוצרים מחלקה נפרדת לכל דף שמכילה את כל ה-locators והפעולות שלו. כך, אם ה-UI משתנה – אנחנו מתקנים רק במחלקת הדף, לא בכל הבדיקות.

🧱 OOP – תכנות מונחה עצמים

כל Page Object הוא מחלקה (class). כל המחלקות יורשות מ-BasePage שמכיל פעולות משותפות (navigate, screenshot). זה עקרון ה-ירושה (Inheritance).

📊 Data-Driven Testing

נתוני הבדיקה (query, maxPrice, credentials) נשמרים בקובץ config/test_data.json נפרד מהקוד. כך אפשר לשנות נתונים בלי לגעת בלוגיקה, ולהריץ אותה בדיקה על פרמטרים שונים.

🏗️ pages/base_page.py – מחלקת הבסיס

כל Page Object יורש ממחלקה זו. היא מכילה את page (אובייקט Playwright) ופעולות משותפות. זה עקרון DRY – Don't Repeat Yourself. 

from playwright.sync_api import Page
import os

class BasePage:
    def __init__(self, page: Page):
        self.page = page           # האובייקט שדרכו שולטים על הדפדפן

    def navigate(self, url: str):
        self.page.goto(url)        # ניווט לכתובת URL

    def take_screenshot(self, name: str):
        os.makedirs("screenshots", exist_ok=True)
        self.page.screenshot(path=f"screenshots/{name}.png")

🏠 pages/home_page.py – דף הבית

מטפל בחיפוש מוצרים. שימו לב ל-locators – ה-CSS selectors שמוצאים אלמנטים בדף. 

from pages.base_page import BasePage

class HomePage(BasePage):
    # ── Locators ────────────────────────────────
    SEARCH_INPUT  = "input[name='search']"
    SEARCH_BUTTON = "#search button[type='button']"
    BASE_URL      = "https://tutorialsninja.com/demo/"

    # ── Methods ─────────────────────────────────
    def open(self):
        self.navigate(self.BASE_URL)
        self.page.wait_for_load_state("networkidle")   # מחכה שהדף ייטען

    def search(self, query: str):
        self.page.fill(self.SEARCH_INPUT, query)        # מקליד בשדה החיפוש
        self.page.click(self.SEARCH_BUTTON)             # לוחץ על כפתור החיפוש
        self.page.wait_for_load_state("networkidle")

🔐 pages/login_page.py – כניסה לאתר

מכיל לוגיקה של Login ו-Register. הפונקציה login_or_register מנסה להיכנס – ואם נכשלת (המשתמש לא קיים), רושמת חדש. 

import time
from pages.base_page import BasePage

class LoginPage(BasePage):
    LOGIN_URL    = "https://tutorialsninja.com/demo/index.php?route=account/login"
    REGISTER_URL = "https://tutorialsninja.com/demo/index.php?route=account/register"

    def login(self, email: str, password: str) -> bool:
        self.navigate(self.LOGIN_URL)
        self.page.fill("#input-email", email)
        self.page.fill("#input-password", password)
        self.page.click("input[type='submit'][value='Login']")
        self.page.wait_for_load_state("networkidle")
        # בדיקה שהכניסה הצליחה – אם יש קישור Logout בתפריט
        return self.page.locator("a[href*='account/logout']").count() > 0

    def register(self, email: str, password: str) -> bool:
        self.navigate(self.REGISTER_URL)
        self.page.fill("#input-firstname", "Test")
        self.page.fill("#input-lastname",  "User")
        self.page.fill("#input-email",     email)
        self.page.fill("#input-telephone", "0501234567")
        self.page.fill("#input-password",  password)
        self.page.fill("#input-confirm",   password)
        self.page.check("input[name='agree']")   # הסכמה לתנאים
        self.page.click("input[value='Continue']")
        self.page.wait_for_load_state("networkidle")
        return "success" in self.page.url

    def login_or_register(self, email: str, password: str):
        if not self.login(email, password):
            # כניסה נכשלה – ניצור חשבון חדש עם אימייל ייחודי
            unique_email = f"test_{int(time.time())}@auto.com"
            self.register(unique_email, password)
            self.login(unique_email, password)

🔍 pages/search_results_page.py – תוצאות חיפוש

הפונקציה המרכזית כאן: get_items_under_price.
היא עוברת על כל המוצרים בדף, מסננת לפי מחיר, ומטפלת ב-Pagination (מעבר לדף הבא אם צריך עוד מוצרים). 

import re
from pages.base_page import BasePage
from utils.price_parser import parse_price

class SearchResultsPage(BasePage):
    PRODUCT_CARD = "div.product-layout"
    PRODUCT_LINK = "div.caption h4 a"
    PRICE        = "p.price"
    PRICE_NEW    = "span.price-new"   # מחיר אחרי הנחה

    def get_items_under_price(self, max_price: float, limit: int = 5) -> list[str]:
        urls = []

        while len(urls) < limit:
            # קבל את כל כרטיסי המוצרים בדף הנוכחי
            cards = self.page.locator(self.PRODUCT_CARD).all()

            for card in cards:
                if len(urls) >= limit:
                    break

                # נסה לקרוא מחיר הנחה, אחרת מחיר רגיל
                price_el = card.locator(self.PRICE_NEW)
                price_text = (price_el.first.inner_text()
                              if price_el.count() > 0
                              else card.locator(self.PRICE).first.inner_text())

                price = parse_price(price_text)   # "$110.00" → 110.0

                if 0 < price <= max_price:
                    href = card.locator(self.PRODUCT_LINK).first.get_attribute("href")
                    if href:
                        urls.append(href)

            # האם יש דף הבא? מחפש כפתור ">" ב-pagination
            next_btn = None
            for a in self.page.locator("ul.pagination a").all():
                if a.inner_text().strip() == ">":
                    next_btn = a
                    break

            if next_btn:
                next_btn.click()
                self.page.wait_for_load_state("networkidle")
            else:
                break   # אין יותר עמודים

        return urls[:limit]

🛍️ pages/product_page.py – דף מוצר

נכנסים לכל URL, בוחרים וריאנטים אם קיימים (dropdown/radio) ולוחצים "Add to Cart". מחכים להופעת הודעת הצלחה ושומרים צילום מסך. 

import random
from pages.base_page import BasePage

class ProductPage(BasePage):
    ADD_TO_CART   = "#button-cart"
    SUCCESS_ALERT = "#alert .alert-success"

    def _select_variants(self):
        # בחירה אקראית מתוך Dropdowns
        for sel in self.page.locator("select[name^='option']").all():
            opts = sel.locator("option[value!='']").all()
            if opts:
                sel.select_option(random.choice(opts).get_attribute("value"))

        # בחירה ראשונה מתוך Radio buttons
        seen = set()
        for radio in self.page.locator("input[type='radio'][name^='option']").all():
            name = radio.get_attribute("name")
            if name not in seen:
                seen.add(name)
                radio.check()

    def add_to_cart(self, url: str) -> bool:
        self.navigate(url)
        self.page.wait_for_load_state("networkidle")
        self._select_variants()
        self.page.locator(self.ADD_TO_CART).click()

        try:
            self.page.wait_for_selector(self.SUCCESS_ALERT, timeout=5000)
            self.take_screenshot(f"added_{hash(url) % 9999}")
            return True
        except:
            self.take_screenshot(f"failed_{hash(url) % 9999}")
            return False

🛒 pages/cart_page.py – עגלת קניות

ניווט לדף העגלה, קריאת שורת ה-"Total" מהטבלה, ופרסור הסכום. 

from pages.base_page import BasePage
from utils.price_parser import parse_price

class CartPage(BasePage):
    URL = "https://tutorialsninja.com/demo/index.php?route=checkout/cart"

    def get_total(self) -> float:
        self.navigate(self.URL)
        self.page.wait_for_load_state("networkidle")
        self.take_screenshot("cart_page")

        # מחפשים את שורת "Total" בטבלת הסיכום
        for row in self.page.locator("table tr").all():
            cells = row.locator("td").all()
            if len(cells) >= 2:
                label = cells[0].inner_text().strip().lower()
                if label == "total":          # "total" ולא "sub-total"
                    return parse_price(cells[-1].inner_text())
        return 0.0

🔢 utils/price_parser.py – פרסור מחירים

ממיר טקסט מחיר מהדף (כגון "$110.00 Ex Tax: $100.00") למספר עשרוני. משמש בכל דף שצריך לקרוא מחיר. 

import re

def parse_price(text: str) -> float:
    """
    "$110.00 Ex Tax: $100.00"  →  110.0
    "$1,299.00"                →  1299.0
    """
    match = re.search(r'\$?([\d,]+\.?\d*)', text.split('\n')[0])
    if match:
        return float(match.group(1).replace(',', ''))
    return 0.0

✅ tests/test_shopping.py – הבדיקה הראשית

כל 4 הפונקציות ממומשות כאן. הבדיקה עוברת את כל תרחיש הקנייה מקצה לקצה. שימו לב לדקורטורים של Allure שמייצרים תיעוד עשיר בדוח. 

import allure
import pytest
from playwright.sync_api import Page
from pages.home_page import HomePage
from pages.login_page import LoginPage
from pages.search_results_page import SearchResultsPage
from pages.product_page import ProductPage
from pages.cart_page import CartPage

# ── 1. הזדהות ──────────────────────────────────────────
def login(page: Page, email: str, password: str):
    LoginPage(page).login_or_register(email, password)

# ── 2. חיפוש לפי מחיר ─────────────────────────────────
def search_items_by_name_under_price(
    page: Page, query: str, max_price: float, limit: int = 5
) -> list[str]:
    home = HomePage(page)
    home.open()
    home.search(query)
    return SearchResultsPage(page).get_items_under_price(max_price, limit)

# ── 3. הוספה לסל ──────────────────────────────────────
def add_items_to_cart(page: Page, urls: list[str]):
    product = ProductPage(page)
    for url in urls:
        with allure.step(f"Adding: {url}"):
            product.add_to_cart(url)

# ── 4. אימות סכום ─────────────────────────────────────
def assert_cart_total_not_exceeds(
    page: Page, budget_per_item: float, items_count: int
):
    total = CartPage(page).get_total()
    threshold = budget_per_item * items_count
    assert total <= threshold, \
        f"Cart total ${total:.2f} exceeds budget ${threshold:.2f}"

# ── בדיקת E2E ─────────────────────────────────────────
class TestShoppingFlow:

    @allure.title("Full E2E Shopping Flow")
    @allure.description("Search → Add to Cart → Assert Total")
    def test_full_flow(self, page: Page, test_data: dict):
        creds  = test_data["credentials"]
        search = test_data["search"]

        with allure.step("שלב 1: הזדהות"):
            login(page, creds["email"], creds["password"])

        with allure.step("שלב 2: חיפוש מוצרים"):
            urls = search_items_by_name_under_price(
                page, search["query"], search["max_price"], search["limit"]
            )
            assert len(urls) > 0, "לא נמצאו מוצרים במחיר המבוקש"

        with allure.step("שלב 3: הוספה לסל"):
            add_items_to_cart(page, urls)

        with allure.step("שלב 4: אימות סכום"):
            assert_cart_total_not_exceeds(page, search["max_price"], len(urls))

⚙️ tests/conftest.py – הגדרות Pytest

Fixtures הן פונקציות שמכינות תנאים לפני הבדיקה. Pytest-Playwright כבר מספק את page fixture – כאן אנחנו מרחיבים אותו. 

import pytest, json, allure
from pathlib import Path

@pytest.fixture(scope="session")    # נטען פעם אחת לכל session
def test_data():
    with open(Path("config/test_data.json"), encoding="utf-8") as f:
        return json.load(f)

@pytest.fixture(scope="session")
def browser_context_args(browser_context_args):
    return {**browser_context_args, "viewport": {"width": 1280, "height": 800}}

# שמירת צילום מסך בכישלון
@pytest.hookimpl(tryfirst=True, hookwrapper=True)
def pytest_runtest_makereport(item, call):
    outcome = yield
    rep = outcome.get_result()
    setattr(item, f"rep_{rep.when}", rep)

@pytest.fixture(autouse=True)   # רץ אוטומטית לכל בדיקה
def screenshot_on_failure(page, request):
    yield
    if getattr(request.node, "rep_call", None) and request.node.rep_call.failed:
        path = f"screenshots/FAIL_{request.node.name}.png"
        page.screenshot(path=path)
        allure.attach.file(path, attachment_type=allure.attachment_type.PNG)

🌍 תיקון 2 – ENV Variables + Profiles (15% Data-Driven)

# conftest.py – קריאת ENV variables שמדרסות את ה-JSON
import os

@pytest.fixture(scope="session")
def test_data():
    with open("config/test_data.json", encoding="utf-8") as f:
        data = json.load(f)

    # ENV override – אפשר להריץ: TEST_EMAIL=me@test.com pytest
    data["base_url"]                 = os.getenv("BASE_URL",       data["base_url"])
    data["credentials"]["email"]    = os.getenv("TEST_EMAIL",    data["credentials"]["email"])
    data["credentials"]["password"] = os.getenv("TEST_PASSWORD", data["credentials"]["password"])
    data["search"]["query"]        = os.getenv("SEARCH_QUERY",  data["search"]["query"])
    return data

🎬 תיקון 3 – Playwright Trace (דרישה מפורשת במטלה)

שורה 51 במטלה: "שמור Screenshot/Trace של עמוד הסל"
Playwright Trace הוא קובץ ZIP שמכיל הקלטה מלאה של הבדיקה – כולל snapshots, network requests ו-console logs. אפשר לפתוח אותו ב-browser. 

# conftest.py – Fixture שמפעיל Trace אוטומטית לכל בדיקה
@pytest.fixture(autouse=True)
def tracing(context, request):
    context.tracing.start(
        screenshots=True,   # צילום מסך בכל פעולה
        snapshots=True,      # snapshot של ה-DOM
        sources=True         # קוד המקור
    )
    yield
    trace_path = f"reports/traces/{request.node.name}.zip"
    Path("reports/traces").mkdir(parents=True, exist_ok=True)
    context.tracing.stop(path=trace_path)
    allure.attach.file(trace_path, name="playwright-trace",
                       attachment_type=allure.attachment_type.ZIP)

לפתיחת ה-Trace אחרי הריצה:

playwright show-trace reports\traces\test_full_flow.zip
