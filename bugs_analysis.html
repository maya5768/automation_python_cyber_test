from playwright.sync_api import sync_playwright
from selenium import webdriver     # ← באג 1 🔴
import time

def test_search_functionality():
    browser = sync_playwright().start().chromium.launch()  # ← באג 2 🔴
    page = browser.new_page()
    page.goto("https://example.com")

    time.sleep(2)                                              # ← באג 3 🟠

    search_box = page.locator("#search")
    search_box.fill("playwright testing")

    page.locator(".button").click()

    browser.close()                                            # ← באג 5 🔴
קריטי 🔴 באג 1 – ייבוא Selenium מיותר ולא בשימוש
שורה 2: from selenium import webdriver

webdriver מיובא אך לא משמש בשום מקום בקוד. זהו dead code – קוד מת שלא תורם כלום.

למה זה בעיה?

  • מערבב שתי ספריות שונות לחלוטין (Playwright ו-Selenium) – מבלבל כל מי שקורא את הקוד
  • יגרום ל-ModuleNotFoundError בסביבה שבה Selenium לא מותקן
  • מסמן לקורא שהקוד "עובד עם Selenium" כשלמעשה לא
✅ תיקון

למחוק את השורה לחלוטין:

# להסיר:
from selenium import webdriver   # ← שורה זו נמחקת

# להשאיר רק:
from playwright.sync_api import sync_playwright
קריטי 🔴 באג 2 – דליפת משאבים: Playwright לא נסגר
שורה 5: browser = sync_playwright().start().chromium.launch()

sync_playwright().start() מחזיר אובייקט Playwright שחייב להיסגר בסיום עם .stop(). כאן האובייקט לא נשמר במשתנה – לכן לעולם לא ייסגר.

מה קורה בפועל?

  • תהליכי דפדפן (Chrome/Chromium) נשארים פעילים ברקע
  • צריכת זיכרון עולה עם כל הרצה
  • בסביבת CI/CD – build agents יכולים להיתקע
✅ תיקון – שימוש ב-context manager (מומלץ)
# הדרך הנכונה – with מבטיח סגירה אוטומטית תמיד
with sync_playwright() as p:
    browser = p.chromium.launch()
    page = browser.new_page()
    # ... כל הקוד כאן ...
    browser.close()
# p.stop() נקרא אוטומטית כשיוצאים מה-with

או לחלופין – שמירה במשתנה וסגירה ידנית:

playwright_instance = sync_playwright().start()
browser = playwright_instance.chromium.launch()
try:
    # ... קוד ...
finally:
    browser.close()
    playwright_instance.stop()   # ← חובה!
אזהרה 🟠 באג 3 – שימוש ב-time.sleep() (Anti-Pattern)
שורות 9 ו-15: time.sleep(2) | time.sleep(3)

Playwright כולל מנגנון המתנה אוטומטי – הוא ממתין שאלמנטים יהיו גלויים ומוכנים לפעולה לפני כל fill(), click() וכו'. time.sleep() הוא המתנה עיוורת שלא יודעת מתי הדף באמת מוכן.

הבעיות:

  • בדיקות שבירות (Flaky Tests) – אם הדף לוקח 2.1 שניות, ה-sleep(2) ייכשל
  • בדיקות איטיות – ממתינים 5 שניות כשהדף נטען ב-200ms
  • מסתיר בעיות אמיתיות – מסנכרן בכח במקום לתקן את שורש הבעיה
✅ תיקון – שימוש ב-Playwright waiters
# במקום: time.sleep(2)
page.wait_for_load_state("networkidle")        # ממתין שאין בקשות רשת

# במקום: time.sleep(3) אחרי לחיצה
page.wait_for_selector(".result-item")         # ממתין לאלמנט ספציפי

# או: Playwright ממתין אוטומטית – לא צריך כלום!
page.locator(".button").click()              # כבר ממתין שהכפתור קיים
לוגי 🟡 באג 4 – אין Assert – הבדיקה לא מאמתת כלום
שורה 18: results = page.locator(".result-item")

page.locator() מחזיר אובייקט Locator בלבד – הוא לא מחפש אלמנטים עדיין (lazy evaluation). בנוסף, results מוגדר אך לא נעשה בו שום שימוש ואין בקוד אפילו assert אחד.

התוצאה: הבדיקה תמיד תסמן PASS – גם אם החיפוש החזיר 0 תוצאות, גם אם האתר קרס.

✅ תיקון – הוספת assert משמעותי
results = page.locator(".result-item")

# אפשרות 1 – assert רגיל
assert results.count() > 0, "Expected search results but found none"

# אפשרות 2 – Playwright expect API (מומלץ)
from playwright.sync_api import expect
expect(results).to_have_count(10)              # מאמת כמות מדויקת
expect(results.first).to_be_visible()           # מאמת שהראשון גלוי
קריטי 🔴 באג 5 – browser.close() בלי try/finally
שורה 20: browser.close()

אם כל שלב לפני זה זורק Exception (לדוגמה: page.locator("#search") לא מוצא את האלמנט), הקוד יקרוס לפני ש-browser.close() ירוץ. הדפדפן ישאר פתוח ויבזבז זיכרון וזמן CPU.

שימו לב שבאג זה מחמיר את באג 2 – לא רק ש-Playwright לא נסגר, גם ה-browser עצמו לא נסגר במקרה של כישלון.

✅ תיקון – try/finally מבטיח סגירה תמיד
browser = playwright_instance.chromium.launch()
try:
    page = browser.new_page()
    page.goto("https://example.com")
    # ... שאר הקוד ...
finally:
    browser.close()    # ירוץ תמיד – גם בכישלון

הפתרון הטוב ביותר: שימוש ב-context manager כמו בתיקון באג 2 – הוא פותר את באגים 2 ו-5 בבת אחת.
